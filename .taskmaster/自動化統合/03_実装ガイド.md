### 1. task_master_adapter.py

Task Master tasks.jsonの読み取り専用アダプター:

```python
"""
Task Master tasks.json読み取り専用アダプター

セキュリティ設計:
- Read-only操作のみ
- tasks.jsonへの書き込み禁止
- task_keyの抽出のみ実行
"""

import json
from pathlib import Path
from typing import Dict, List, Optional
from dataclasses import dataclass


@dataclass
class TaskMasterTask:
    """Task Masterタスク表現"""
    id: str
    title: str
    status: str
    task_key: Optional[str] = None  # allowed_tasks.ymlのキー
    priority: str = "medium"
    dependencies: List[str] = None

    def __post_init__(self):
        if self.dependencies is None:
            self.dependencies = []


class TaskMasterAdapter:
    """Task Master tasks.json読み取り専用アダプター"""

    def __init__(self, tasks_json_path: str = ".taskmaster/tasks/tasks.json"):
        self.tasks_path = Path(tasks_json_path)
        if not self.tasks_path.exists():
            raise FileNotFoundError(f"tasks.json not found: {self.tasks_path}")

    def read_tasks(self) -> List[TaskMasterTask]:
        """
        tasks.jsonから全タスク読み取り

        Returns:
            List[TaskMasterTask]: タスク一覧

        Raises:
            json.JSONDecodeError: JSON形式エラー
            PermissionError: 読み取り権限エラー
        """
        try:
            with open(self.tasks_path, 'r', encoding='utf-8') as f:
                data = json.load(f)

            tasks = []
            # タグ別タスク構造をフラット化
            for tag_data in data.get("tags", {}).values():
                for task_dict in tag_data.get("tasks", []):
                    task = self._parse_task(task_dict)
                    if task:
                        tasks.append(task)

            return tasks

        except json.JSONDecodeError as e:
            raise ValueError(f"Invalid JSON format: {e}")
        except PermissionError:
            raise PermissionError(f"Cannot read tasks.json: {self.tasks_path}")

    def get_pending_tasks(self) -> List[TaskMasterTask]:
        """
        pending状態のタスクのみ取得

        Returns:
            List[TaskMasterTask]: pending状態タスク一覧
        """
        all_tasks = self.read_tasks()
        return [task for task in all_tasks if task.status == "pending"]

    def get_executable_tasks(self) -> List[TaskMasterTask]:
        """
        実行可能タスク（task_keyあり + pending）取得

        Returns:
            List[TaskMasterTask]: 実行可能タスク一覧
        """
        pending_tasks = self.get_pending_tasks()
        return [
            task for task in pending_tasks
            if task.task_key is not None
        ]

    def _parse_task(self, task_dict: Dict) -> Optional[TaskMasterTask]:
        """
        タスク辞書をTaskMasterTaskオブジェクトに変換

        Args:
            task_dict: tasks.jsonのタスク辞書

        Returns:
            TaskMasterTask or None: 変換されたタスク
        """
        try:
            # task_keyの抽出（details内のメタデータから）
            details = task_dict.get("details", "")
            task_key = self._extract_task_key(details)

            return TaskMasterTask(
                id=task_dict["id"],
                title=task_dict["title"],
                status=task_dict["status"],
                task_key=task_key,
                priority=task_dict.get("priority", "medium"),
                dependencies=task_dict.get("dependencies", [])
            )
        except KeyError as e:
            # 必須フィールド欠損時はスキップ
            return None

    @staticmethod
    def _extract_task_key(details: str) -> Optional[str]:
        """
        タスク詳細からtask_keyを抽出

        Args:
            details: タスク詳細文字列

        Returns:
            str or None: 抽出されたtask_key

        Example:
            詳細: "pytest実行: task_key=run_unit_tests"
            → "run_unit_tests"
        """
        import re

        # "task_key=XXX" パターンを検索
        match = re.search(r'task_key=([a-zA-Z0-9_-]+)', details)
        if match:
            return match.group(1)

        return None


# 使用例
if __name__ == "__main__":
    adapter = TaskMasterAdapter()

    # 実行可能タスク取得
    executable_tasks = adapter.get_executable_tasks()
    for task in executable_tasks:
        print(f"Task {task.id}: {task.title} (key: {task.task_key})")
```

### 2. security_validator.py

2段階セキュリティ検証モジュール:

```python
"""
2段階セキュリティ検証モジュール

検証レイヤー:
1. task_key形式検証
2. allowed_tasks.ymlホワイトリスト検証
3. ソース別追加検証（Task Master経由時）
"""

import re
import yaml
import logging
from pathlib import Path
from typing import Dict, Optional
from dataclasses import dataclass
from datetime import datetime


@dataclass
class ValidationResult:
    """検証結果"""
    status: str  # "approved" or "rejected"
    command: Optional[str] = None
    timeout: Optional[int] = None
    reason: Optional[str] = None
    source: Optional[str] = None


class SecurityValidator:
    """2段階セキュリティ検証"""

    def __init__(self, allowed_tasks_path: str = "allowed_tasks.yml"):
        self.allowed_tasks_path = Path(allowed_tasks_path)
        if not self.allowed_tasks_path.exists():
            raise FileNotFoundError(f"allowed_tasks.yml not found: {self.allowed_tasks_path}")

        self.allowed_tasks = self._load_allowed_tasks()
        self.logger = self._setup_logger()

    def validate_command(self, task_key: str, source: str) -> ValidationResult:
        """
        コマンド実行前の2段階検証

        Args:
            task_key: 実行するタスクキー
            source: データソース ("tasks.json" or "todo.md")

        Returns:
            ValidationResult: 検証結果
        """
        # Step 1: task_key形式検証
        if not self._is_valid_key_format(task_key):
            self._log_security_event("invalid_format", task_key, source)
            return ValidationResult(
                status="rejected",
                reason=f"Invalid task_key format: {task_key}",
                source=source
            )

        # Step 2: ホワイトリスト検証
        if task_key not in self.allowed_tasks:
            self._log_security_event("unauthorized_access", task_key, source)
            return ValidationResult(
                status="rejected",
                reason=f"Task not in whitelist: {task_key}",
                source=source
            )

        # Step 3: ソース別追加検証
        if source == "tasks.json":
            # Task Master経由の場合、追加検証
            if not self._verify_task_master_integrity(task_key):
                self._log_security_event("integrity_check_failed", task_key, source)
                return ValidationResult(
                    status="rejected",
                    reason="Task Master integrity check failed",
                    source=source
                )

        # 検証成功
        self._log_security_event("approved", task_key, source)
        return ValidationResult(
            status="approved",
            command=self.allowed_tasks[task_key]["command"],
            timeout=self.allowed_tasks[task_key].get("timeout", 300),
            source=source
        )

    @staticmethod
    def _is_valid_key_format(task_key: str) -> bool:
        """
        task_key形式検証

        許可形式: [a-zA-Z0-9_-]+
        禁止文字: スペース、特殊記号（-と_以外）
        """
        return bool(re.match(r'^[a-zA-Z0-9_-]+$', task_key))

    def _verify_task_master_integrity(self, task_key: str) -> bool:
        """
        Task Master経由タスクの整合性検証

        検証内容:
        - tasks.jsonの存在確認
        - タスクキーのメタデータ確認
        """
        # 実装例: tasks.jsonが改ざんされていないか確認
        # 本番環境ではファイルハッシュ検証等を追加
        tasks_json_path = Path(".taskmaster/tasks/tasks.json")
        return tasks_json_path.exists()

    def _load_allowed_tasks(self) -> Dict:
        """allowed_tasks.yml読み込み"""
        try:
            with open(self.allowed_tasks_path, 'r', encoding='utf-8') as f:
                return yaml.safe_load(f)
        except yaml.YAMLError as e:
            raise ValueError(f"Invalid YAML format: {e}")

    @staticmethod
    def _setup_logger() -> logging.Logger:
        """セキュリティログ設定"""
        logger = logging.getLogger("SecurityValidator")
        logger.setLevel(logging.INFO)

        # ファイルハンドラ
        handler = logging.FileHandler("logs/security_audit.log")
        formatter = logging.Formatter(
            '%(asctime)s - %(levelname)s - %(message)s'
        )
        handler.setFormatter(formatter)
        logger.addHandler(handler)

        return logger

    def _log_security_event(self, event_type: str, task_key: str, source: str):
        """セキュリティイベントログ"""
        self.logger.info(
            f"[{event_type}] task_key={task_key}, source={source}, "
            f"timestamp={datetime.now().isoformat()}"
        )


# 使用例
if __name__ == "__main__":
    validator = SecurityValidator()

    # todo.md経由の検証
    result1 = validator.validate_command("run_unit_tests", source="todo.md")
    print(f"Result: {result1.status}, Command: {result1.command}")

    # Task Master経由の検証
    result2 = validator.validate_command("run_unit_tests", source="tasks.json")
    print(f"Result: {result2.status}, Command: {result2.command}")

    # 不正なキーの検証
    result3 = validator.validate_command("rm -rf /", source="todo.md")
    print(f"Result: {result3.status}, Reason: {result3.reason}")
```

### 3. task_master_executor.py

統合実行ロジック:

```python
"""
Task Master統合実行ロジック

データソース優先順位:
1. todo.md (YAML Front Matter) - 優先
2. Task Master tasks.json - 補助
"""

import subprocess
import shlex
from pathlib import Path
from typing import List, Dict, Optional
from dataclasses import dataclass

from task_master_adapter import TaskMasterAdapter, TaskMasterTask
from security_validator import SecurityValidator, ValidationResult


@dataclass
class ExecutionResult:
    """実行結果"""
    task_id: str
    task_key: str
    status: str  # "success" or "failed"
    output: Optional[str] = None
    error: Optional[str] = None
    exit_code: Optional[int] = None


class TaskMasterExecutor:
    """Task Master統合実行エンジン"""

    def __init__(
        self,
        adapter: TaskMasterAdapter,
        validator: SecurityValidator
    ):
        self.adapter = adapter
        self.validator = validator

    def execute_task_master_tasks(self) -> List[ExecutionResult]:
        """
        Task Master tasks.jsonから実行可能タスクを実行

        Returns:
            List[ExecutionResult]: 実行結果一覧
        """
        # Step 1: 実行可能タスク取得
        executable_tasks = self.adapter.get_executable_tasks()

        if not executable_tasks:
            print("No executable tasks found in tasks.json")
            return []

        # Step 2: 各タスクを実行
        results = []
        for task in executable_tasks:
            result = self._execute_single_task(task)
            results.append(result)

        return results

    def _execute_single_task(self, task: TaskMasterTask) -> ExecutionResult:
        """
        単一タスク実行

        Args:
            task: TaskMasterTaskオブジェクト

        Returns:
            ExecutionResult: 実行結果
        """
        # Step 1: セキュリティ検証
        validation = self.validator.validate_command(
            task.task_key,
            source="tasks.json"
        )

        if validation.status == "rejected":
            return ExecutionResult(
                task_id=task.id,
                task_key=task.task_key,
                status="failed",
                error=f"Security validation failed: {validation.reason}"
            )

        # Step 2: コマンド実行
        try:
            result = subprocess.run(
                shlex.split(validation.command),
                timeout=validation.timeout,
                shell=False,  # 必須: shell injection防止
                capture_output=True,
                text=True
            )

            return ExecutionResult(
                task_id=task.id,
                task_key=task.task_key,
                status="success" if result.returncode == 0 else "failed",
                output=result.stdout,
                error=result.stderr if result.returncode != 0 else None,
                exit_code=result.returncode
            )

        except subprocess.TimeoutExpired:
            return ExecutionResult(
                task_id=task.id,
                task_key=task.task_key,
                status="failed",
                error=f"Command timeout ({validation.timeout}s)"
            )
        except Exception as e:
            return ExecutionResult(
                task_id=task.id,
                task_key=task.task_key,
                status="failed",
                error=f"Execution error: {str(e)}"
            )


# CLI実行例
if __name__ == "__main__":
    import sys

    # 初期化
    adapter = TaskMasterAdapter()
    validator = SecurityValidator()
    executor = TaskMasterExecutor(adapter, validator)

    # 実行
    print("Executing Task Master tasks...")
    results = executor.execute_task_master_tasks()

    # 結果表示
    for result in results:
        print(f"\n{'='*60}")
        print(f"Task ID: {result.task_id}")
        print(f"Task Key: {result.task_key}")
        print(f"Status: {result.status}")
        if result.output:
            print(f"Output:\n{result.output}")
        if result.error:
            print(f"Error:\n{result.error}")

    # 終了コード
    failed_count = sum(1 for r in results if r.status == "failed")
    sys.exit(1 if failed_count > 0 else 0)
```

# allowed_tasks.ymlに追加
new_task_key:

long_running_task:
  command: "pytest tests/integration/"

# task_master_adapter.py にキャッシング追加
from functools import lru_cache

class TaskMasterAdapter:
