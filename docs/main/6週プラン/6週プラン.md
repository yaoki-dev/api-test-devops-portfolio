class APIClientError(Exception):
    """基底例外クラス"""
    pass

class APIHTTPError(APIClientError):
    """HTTPエラー（4xx/5xx）"""
    def __init__(self, status_code: int, message: str):
        self.status_code = status_code
        super().__init__(f"HTTP {status_code}: {message}")

class APIConnectionError(APIClientError):
    """接続エラー"""
    pass

class APITimeoutError(APIClientError):
    """タイムアウトエラー"""
    pass

class APIRetryError(APIClientError):
    """リトライ上限エラー"""
    pass

# utils/api_client.py - リトライロジック実装
from random import uniform  # PEP 8: importはモジュール先頭に配置

def _make_request_with_retry(self, method: str, endpoint: str, **kwargs: Any) -> httpx.Response:
    """リトライロジック付きリクエスト実行

    - 5xxエラー: 3回リトライ（Exponential backoff）
    - 4xxエラー: 即座に失敗（リトライなし）
    """
    for attempt in range(self.retry_count):
        try:
            response = self._client.request(method, endpoint, **kwargs)
            response.raise_for_status()
            return response
        except httpx.HTTPStatusError as e:
            if e.response.status_code < 500:
                # 4xxエラー: リトライせず即座に失敗（例外チェーン保持）
                raise APIHTTPError(e.response.status_code, str(e)) from e
            if attempt == self.retry_count - 1:
                # 最終試行で失敗
                raise APIRetryError(f"Max retries ({self.retry_count}) exceeded") from e
            # Exponential backoff + jitter（サーバー負荷分散）
            base_delay = self.retry_delay * (2 ** attempt)
            jitter = uniform(0, 1)  # 固定ジッター（シンプル化）
            # 実務では比例ジッター（base_delay * uniform(0, 0.3)）も検討
            time.sleep(min(base_delay + jitter, 10.0))  # max 10秒
        except httpx.ConnectError as e:
            if attempt == self.retry_count - 1:
                raise APIConnectionError(str(e)) from e
            # 一時的なネットワーク障害 → リトライ
            base_delay = self.retry_delay * (2 ** attempt)
            jitter = uniform(0, 1)
            time.sleep(min(base_delay + jitter, 10.0))
        except httpx.TimeoutException as e:
            if attempt == self.retry_count - 1:
                raise APITimeoutError(str(e)) from e
            # タイムアウト → リトライ（ネットワーク遅延は一時的）
            base_delay = self.retry_delay * (2 ** attempt)
            jitter = uniform(0, 1)
            time.sleep(min(base_delay + jitter, 10.0))
```

# clients/jsonplaceholder_client.py - 専用クライアント作成
class JSONPlaceholderClient(BaseAPIClient):
    """JSONPlaceholder API専用クライアント"""

    def __init__(self):
        super().__init__("https://jsonplaceholder.typicode.com")

    def get_user(self, user_id: int) -> dict:
        """ユーザー情報取得

        Args:
            user_id: ユーザーID（正の整数）

        Returns:
            ユーザー情報辞書

        Raises:
            ValueError: user_id <= 0の場合
            APIHTTPError: 4xxエラー（存在しないユーザーなど）
            APIRetryError: リトライ上限到達
        """
        if user_id <= 0:
            raise ValueError(f"user_id must be positive, got {user_id}")
        return self.get(f"/users/{user_id}")

    def get_posts(self, user_id: int | None = None) -> list[dict]:
        """投稿一覧取得

        Args:
            user_id: フィルタ用ユーザーID（省略時は全投稿）

        Returns:
            投稿情報辞書のリスト
        """
        params = {"userId": user_id} if user_id else None
        return self.get("/posts", params=params)
```

# ※概念理解用サンプル。本格実装はWeek 5（非同期処理深化）で詳細化
from enum import Enum
from time import time

class CircuitState(Enum):
    CLOSED = "closed"      # 正常動作
    OPEN = "open"          # 遮断中（リクエスト拒否）
    HALF_OPEN = "half_open"  # 回復試行中

class CircuitBreaker:
    """サーキットブレーカー（簡易版）

    状態遷移: CLOSED → OPEN → HALF_OPEN → CLOSED
    連続失敗がthreshold到達 → OPEN（reset_timeout秒間リクエスト拒否）
    """
    def __init__(self, failure_threshold: int = 3, reset_timeout: float = 30.0):
        self.failure_threshold = failure_threshold
        self.reset_timeout = reset_timeout
        self._state = CircuitState.CLOSED
        self._failure_count = 0
        self._last_failure_time: float | None = None

    @property
    def state(self) -> CircuitState:
        # OPEN状態でtimeout経過 → HALF_OPENへ遷移
        if self._state == CircuitState.OPEN:
            if self._last_failure_time and time() - self._last_failure_time > self.reset_timeout:
                self._state = CircuitState.HALF_OPEN
        return self._state

    def record_success(self) -> None:
        """成功記録 → CLOSEDへ遷移"""
        self._failure_count = 0
        self._state = CircuitState.CLOSED

    def record_failure(self) -> None:
        """失敗記録 → threshold到達でOPENへ遷移"""
        self._failure_count += 1
        self._last_failure_time = time()
        if self._failure_count >= self.failure_threshold:
            self._state = CircuitState.OPEN

from pydantic import SecretStr, field_validator
from pydantic_settings import BaseSettings, SettingsConfigDict

class APIConfig(BaseSettings):
    """API設定"""
    base_url: str
    timeout: float = 30.0
    retry_count: int = 3
    retry_delay: float = 1.0

    model_config = SettingsConfigDict(
        env_prefix="API__",
        env_file=".env",
        env_file_encoding="utf-8"
    )

class SecurityConfig(BaseSettings):
    """セキュリティ設定"""
    api_key: SecretStr
    rate_limit_requests: int = 100

    model_config = SettingsConfigDict(
        env_prefix="SECURITY__",
        env_file=".env"
    )

class LogConfig(BaseSettings):
    """ログ設定"""
    level: str = "INFO"
    format: str = "console"  # console | json

    model_config = SettingsConfigDict(
        env_prefix="LOG__",
        env_file=".env"
    )

class Settings(BaseSettings):
    """統合設定"""
    environment: str = "development"
    debug: bool = False

    # Pydantic v2: 子設定はデフォルト値なしで宣言（env読み込み時に自動初期化）
    api: APIConfig
    security: SecurityConfig
    log: LogConfig

    @field_validator("environment")
    @classmethod
    def validate_environment(cls, v: str) -> str:
        allowed = ["development", "testing", "staging", "production"]
        if v not in allowed:
            raise ValueError(f"environment must be one of {allowed}")
        return v

settings = Settings()

**Trivy実装詳細**（DevOps推奨版）:

```yaml
# .github/workflows/security-scan.yml
name: Security Scan
on:
  push:
    branches: [main]
  pull_request:
    branches: [main]
  schedule:
    - cron: '0 3 * * 1'  # 毎週月曜10時ICT（タイ時間）= UTC 03:00

jobs:
  trivy-scan:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Build production image
        run: docker build --target runtime -t myapp:${{ github.sha }} .

      - name: Trivy image scan (CRITICAL)
        uses: aquasecurity/trivy-action@0.28.0
        with:
          image-ref: 'myapp:${{ github.sha }}'
          severity: 'CRITICAL'
          exit-code: '1'
          format: 'sarif'
          output: 'trivy-results.sarif'

      - name: Trivy FS scan (CRITICAL - blocking)
        run: trivy fs --severity CRITICAL --exit-code 1 .

      - name: Trivy FS scan (HIGH - warning)
        run: trivy fs --severity HIGH --exit-code 0 .

      - name: Upload results to GitHub Security
        uses: github/codeql-action/upload-sarif@v3
        if: always()
        with:
          sarif_file: 'trivy-results.sarif'
```

```

class AsyncJSONPlaceholderClient(AsyncAPIClient):
    async def get_multiple_users(self, user_ids: list[int]) -> list[dict]:
        """複数ユーザー並行取得"""
        tasks = [self.get_user(user_id) for user_id in user_ids]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        # エラー処理
        users = []
        for result in results:
            if isinstance(result, Exception):
                logger.warning("User fetch failed", error=str(result))
            else:
                users.append(result)
        return users
        # AI説明 → gather + error handling理解

    async def get_user_complete_data(self, user_id: int) -> dict:
        """ユーザー完全データ並行取得"""
        user, posts, todos = await asyncio.gather(
            self.get_user(user_id),
            self.get_posts(user_id),
            self.get_todos(user_id)
        )
        return {
            "user": user,
            "posts": posts,
            "todos": todos
        }
        # AI説明 → concurrent data fetching理解

# 非同期リトライロジック
async def get_with_retry(
    self,
    endpoint: str,
    retries: int = 3,
    delay: float = 1.0
) -> dict:
    """非同期リトライロジック実装"""
    for attempt in range(retries):
        try:
            return await self.get(endpoint)
        except httpx.TimeoutException:
            if attempt == retries - 1:
                raise APITimeoutError(
                    f"Timeout after {retries} retries"
                )
            await asyncio.sleep(delay * (attempt + 1))
```

# Async Context Manager改善
class AsyncJSONPlaceholderClient(AsyncAPIClient):
    """非同期コンテキストマネージャー実装"""

    async def __aenter__(self):
        """リソース初期化"""
        self._client = httpx.AsyncClient(
            base_url=self.base_url,
            timeout=30.0,
            limits=httpx.Limits(
                max_connections=10,
                max_keepalive_connections=5
            )
        )
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """リソース解放"""
        if self._client:
            await self._client.aclose()
        # 例外が発生してもリソース解放を保証
        return False  # 例外は再送出
        # AI説明 → リソース管理理解

# structlog基礎設定
import structlog

# config/logging_config.py
def setup_logging():
    """structlog設定"""
    structlog.configure(
        processors=[
            structlog.processors.TimeStamper(fmt="iso"),
            structlog.stdlib.add_log_level,
            structlog.processors.JSONRenderer()
        ],
        logger_factory=structlog.PrintLoggerFactory(),
    )

# AsyncAPIClientへのstructlog統合
logger = structlog.get_logger()

async def get(self, endpoint: str) -> dict:
    """structlog統合GET実装"""
    logger.info("api_request_start", endpoint=endpoint)
    try:
        response = await self._client.get(endpoint)
        logger.info("api_request_success",
                   endpoint=endpoint,
                   status=response.status_code,
                   duration=response.elapsed.total_seconds())
        return response.json()
    except Exception as e:
        logger.error("api_request_failed",
                    endpoint=endpoint,
                    error=str(e))
        raise

    """JSONPlaceholder API非同期クライアント完成版"""

    async def get_user(self, user_id: int) -> dict:
        """ユーザー情報取得"""
        return await self.get(f"/users/{user_id}")

    async def get_posts(self, user_id: Optional[int] = None) -> list[dict]:
        """投稿一覧取得"""
        endpoint = f"/posts?userId={user_id}" if user_id else "/posts"
        return await self.get(endpoint)

    async def get_todos(self, user_id: int) -> list[dict]:
        """TODO一覧取得"""

    async def get_user_data(self, user_id: int) -> dict:

        user, posts, todos = await asyncio.gather(
            self.get_user(user_id),
            self.get_posts(user_id),

import httpx
from typing import Optional

class AsyncAPIClient:
    def __init__(
        self,
        base_url: str,
        timeout: float = 30.0,
        max_connections: int = 10,
        max_keepalive_connections: int = 5
    ):
        self.base_url = base_url
        self._client: Optional[httpx.AsyncClient] = None
        self._timeout = timeout
        self._limits = httpx.Limits(
            max_connections=max_connections,
            max_keepalive_connections=max_keepalive_connections
        )
        # AI説明 → connection pooling理解

    async def __aenter__(self):
        self._client = httpx.AsyncClient(
            base_url=self.base_url,
            timeout=self._timeout,
            limits=self._limits
        )
        return self
        # AI説明 → production-ready client理解

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self._client:
            await self._client.aclose()
        # graceful shutdown保証

FROM base AS runtime

COPY . .

RUN find . -type d -name __pycache__ -exec rm -rf {} + && \
    find . -type f -name "*.pyc" -delete && \
    rm -rf .git .pytest_cache
# 非rootユーザー設定（本番セキュリティ）
RUN adduser --disabled-password --gecos '' appuser && \
    chown -R appuser:appuser /app
USER appuser
# デフォルトCMD（アプリケーション起動）
CMD ["python", "-m", "uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]

# Stage 4: Test（テスト実行専用）

USER root
RUN pip install --no-cache-dir pytest pytest-cov
