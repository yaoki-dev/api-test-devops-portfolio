```python
async def create_post(self, title: str, body: str, user_id: int) -> dict[str, Any]:
    """新規投稿の非同期作成（入力値検証付き）"""
    # 入力値検証
    if not title or not title.strip():
        raise ValueError("title must not be empty")
    if not body or not body.strip():
        raise ValueError("body must not be empty")
    if user_id <= 0:
        raise ValueError("user_id must be positive integer")

    data = {"title": title.strip(), "body": body.strip(), "userId": user_id}
    response = await self.post("/posts", json=data)
    return response.json()

async def update_post(self, post_id: int, title: str, body: str) -> dict[str, Any]:
    """投稿更新の非同期実行（入力値検証付き）"""
    if post_id <= 0:
        raise ValueError("post_id must be positive integer")
    if not title or not title.strip():
        raise ValueError("title must not be empty")
    if not body or not body.strip():
        raise ValueError("body must not be empty")

    data = {"title": title.strip(), "body": body.strip()}
    response = await self.put(f"/posts/{post_id}", json=data)
    return response.json()

async def delete_post(self, post_id: int) -> None:
    """投稿削除の非同期実行（入力値検証付き）"""
    if post_id <= 0:
        raise ValueError("post_id must be positive integer")

    await self.delete(f"/posts/{post_id}")
```

#### ❌ 現在の実装（固定待機時間）

```python
if attempt < self.retry_count:
    self.logger.debug(f"Waiting {self.retry_delay} seconds before retry...")
    await asyncio.sleep(self.retry_delay)  # 常に1秒待機
```

#### ✅ 修正案: Exponential Backoff実装

```python
if attempt < self.retry_count:
    # Exponential backoff: 1s, 2s, 4s, 8s...
    backoff_delay = self.retry_delay * (2 ** attempt)
    self.logger.debug(
        f"Retry {attempt + 1}/{self.retry_count + 1}: "
        f"Waiting {backoff_delay:.1f}s (exponential backoff)"
    )
    await asyncio.sleep(backoff_delay)
```

```python
async def get_multiple_users(
    self, user_ids: list[int], return_exceptions: bool = True
) -> list[dict[str, Any] | Exception]:
    """
    複数ユーザー情報を並行取得

    Args:
        user_ids: 取得するユーザーIDのリスト
        return_exceptions: True=例外も結果に含める、False=例外時は全体失敗

    Returns:
        各ユーザー情報またはExceptionのリスト

    Example:
        >>> async with AsyncJSONPlaceholderClient() as client:
        ...     results = await client.get_multiple_users([1, 2, 3])
        ...     for user_or_error in results:
        ...         if isinstance(user_or_error, Exception):
        ...             print(f"Error: {user_or_error}")
        ...         else:
        ...             print(f"User: {user_or_error['name']}")
    """
    import asyncio

    # 入力値検証
    if not user_ids:
        raise ValueError("user_ids must not be empty")
    if any(uid <= 0 for uid in user_ids):
        raise ValueError("All user_ids must be positive integers")

    self.logger.info(f"Fetching {len(user_ids)} users concurrently...")

    # 並行タスク生成
    tasks = [self.get_user(user_id) for user_id in user_ids]

    # 並行実行（return_exceptionsで例外処理制御）
    results = await asyncio.gather(*tasks, return_exceptions=return_exceptions)

    self.logger.info(
        f"Completed fetching {len(user_ids)} users: "
        f"{sum(1 for r in results if not isinstance(r, Exception))} succeeded, "
        f"{sum(1 for r in results if isinstance(r, Exception))} failed"
    )

    return results
```
