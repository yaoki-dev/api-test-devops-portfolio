**改善案**:
1. **パフォーマンステスト拡充**:
   ```python
   # tests/performance/test_async_performance.py
   @pytest.mark.performance
   async def test_concurrent_requests_performance():
       """並行処理効果の定量測定"""
       # 1. 同期版: 10リクエスト
       start = time.time()
       for i in range(1, 11):
           sync_client.get_user(i)
       sync_duration = time.time() - start

       # 2. 非同期版: 10リクエスト
       start = time.time()
       async with AsyncClient() as client:
           await asyncio.gather(
               *[client.get_user(i) for i in range(1, 11)]
           )
       async_duration = time.time() - start

       # 3. 効果測定
       speedup = sync_duration / async_duration
       assert speedup >= 8.0  # 10倍近い高速化期待

       # 4. 結果記録
       print(f"同期: {sync_duration:.2f}s")
       print(f"非同期: {async_duration:.2f}s")
       print(f"高速化率: {speedup:.1f}倍")
   ```
